// app/robots.ts
import type { MetadataRoute } from "next";

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      {
        userAgent: "*",
        allow: "/",
        disallow: [
          "/auth",
          "/dashboard",
          "/tchat",
          "/api",
        ],
      },
    ],
    sitemap: "https://eleveai.fr/sitemap.xml",
  };
}
